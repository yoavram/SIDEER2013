Adaptation on complex fitness landscapes

Some mutations are deleterious alone but beneficial together.

This is the source of an interesting problem in evolution: how can populations shift peaks from a local fitness maximum to a global fitness maximum?

A simple example is a two loci model - A/a and B/b.

There are two peaks, a local one at ab and a global one at AB, but ab is currently the frequent type.

from mpl_toolkits.mplot3d import axes3d
import matplotlib.pyplot as plt
from matplotlib import cm
cmap = cm.OrRd

def plot_fitness_landscape(s, H):
    def fitness(a,b):
        if a == 1 and b == 1: return 1+s*H
        else: return (1-s)**(abs(a)+abs(b))
    aA = arange(-4,5)
    bB = arange(-4,5)
    X,Y = meshgrid(aA,bB)
    Z = array([[fitness(x,y) for y in bB] for x in aA])
    fig = plt.figure(figsize=(18,6))
    ax = fig.add_subplot(1, 2, 1, projection='3d')
    p = ax.plot_surface(X, Y, Z, rstride=1, cstride=1, alpha=0.99, cmap=cmap)
    cset = ax.contour(X, Y, Z, zdir='z', offset=0, cmap=cmap)
    ax.view_init(20, -40)
    ax.set_zlim3d(0,1+s*H);
    ax.set_xlabel("a/A")
    ax.set_ylabel("b/B")
    ax.set_zlabel("Fitness")
    ax.set_xticklabels([])
    ax.set_yticklabels([])
    ax = fig.add_subplot(1, 2, 2)
    im = ax.imshow(Z,cmap=cmap, origin='lower')
    ax.set_xticklabels([])
    ax.set_yticklabels([])
    ax.set_xlabel("a/A")
    ax.set_ylabel("b/B")
    ax.set_xticks=arange(-5,5)
    cb = fig.colorbar(im)
    cb.set_label("Fitness")
    return plt
